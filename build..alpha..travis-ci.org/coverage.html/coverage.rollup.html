const Promise = require('bluebird');
const cheerio = require('cheerio').load;
const request = require('request-promise');

const parseQuery = require('querystring').parse;
 
const AuthError = require('../errors/auth');
const RequestError = require('../errors/request');

const { API_VERSION, USER_AGENT, AUTH_ERRORS } = require('../util/constants');
 
const {
	PAGE_BLOCKED,
	INVALID_PHONE_NUMBER,
	AUTHORIZATION_FAILED
} = AUTH_ERRORS;
 
/**
 *  
 *
 * @public
 */
class StandaloneAuth {
	/**
	 * 
	 *
	 * @param {VK} vk
	 */
	constructor (vk) {
		this.vk = vk;
 
		this.jar = request.jar();
	}
 
	/**
	 *  CookieJar
	 *
	 * @return {CookieJar}
	 */
	getCookieJar () {
		return this.jar;
	}

	 *  CookieJar
	 *   
	 *

	 *

	setCookieJar (jar) {
		this.jar = jar;

	 *  Cookie
	 * vk.com -   
	 * login.vk.com -    
	 *
	 * @return {Object}
	 */
	getCookie () {
		return {
			'login.vk.com': this.jar.getCookieString('https://login.vk.com'),
			'vk.com': this.jar.getCookieString('https://vk.com')
		};
	}
 
	/**
	 *  
	 *
	 * @param {Object} options
	 *
	 * @return {Promise}
	 */
	request (options) {
		return this._request(options)
		.catch((error) => {
			if ('body' in error.response) {
				let body;
 
				try {
					body = JSON.parse(error.response.body);
				} catch (e) {
					throw new RequestError(error);
				}
 
				throw new AuthError({
					message: body.error + ', ' + body.error_description,
					code: AUTHORIZATION_FAILED
				});
			}
 
			throw new RequestError(error);
		});
	}

	 *  
	 *
	 * @return {Promise}
	 */

		this._request = this.vk.request.defaults({
			json: false,
			jar: this.jar,
			followAllRedirects: true,
			resolveWithFullResponse: true,
			headers: {
				'User-Agent': USER_AGENT
			}
		});
 
		return this._getBlank()
		.then((response) => (
			this._route(response)
		))
		.then((response) => {
			const hash = parseQuery((response.request.uri.hash || '').replace(/^#/, ''));
 
			if ('access_token' in hash) {
				return hash.access_token;
			}
 
			throw new AuthError({
				message: 'Failed to get token',
				code: AUTHORIZATION_FAILED
			});
		});
	}

	 *
	 * @param {Object} response
	 * @param {Cheeri} $
	 *
	 * @return {Promise}

	_route (response, $ = cheerio(response.body)) {
		if ($('input[name="pass"]').length !== 0) {
			debug('Parse the authorization form');
 
			return this._parseAuthForm(response, $);
		}
 
		if ($('input[name="code"]').length !== 0) {
			const { action, fields } = parseForm($);
 
			if (action.includes('act=authcheck_code')) {
				debug('Processes the authorization code');
 
				return this._authCheckCode(action, fields);
			}
 
			if (action.includes('act=security_check')) {
				debug('Processes the authorization confirm number');
 
				return this._securityPhoneCheck(response, $);
			}
 
			return Promise.reject(new AuthError({
				message: 'Unknown type of authorization check',
				code: AUTHORIZATION_FAILED
			}));
		}

		if (path && path.includes('act=blocked')) {
			return Promise.reject(new AuthError({
				message: 'Page blocked',
				code: PAGE_BLOCKED
			}));
		}

			});

			message: 'Could not log in',
			code: AUTHORIZATION_FAILED
		}));
	}

		fields.email = login || phone;
		fields.pass = pass;

			uri: action,

	}

const { getMethodApi, getChainCode, resolvePromisesTask } = require('../util/helpers');
 
/**

 * @public
 */
class Chain {
	/**
	 * 
	 *
	 * @param {VK} vk
	 */
	constructor (vk) {
		this.vk = vk;
 
		this._queue = [];
		this._isRun = false;
	}
 
	/**
	 *   execute
	 *
	 * @param {VK}     vk
	 * @param {string} method
	 * @param {Object} params
	 *
	 * @return {Promise}
	 */
	static executes (vk, method, queues) {
		const promises = [];
 
		while (queues.length !== 0) {
			const code = getChainCode(
				queues.splice(0, 25).map((params) => getMethodApi(method, params))
			);
 
			promises.push(
				vk.api.execute({
					code: code
				})
				.then(({ response }) => response)
			);
		}
 
		return Promise.all(promises)
		.then((responses) => Array.prototype.concat.apply([], responses));
	}
 
	/**
	 *    
	 *
	 * @param {string} method
	 * @param {Object} params
	 *
	 * @return {Promise}
	 */
	append (method, params = {}) {
		if (this._isRun) {
			throw new Error('Chain  !');
		}
 
		return new Promise((resolve, reject) => {
			this._queue.push({
				method: getMethodApi(method, params),
				resolve,
				reject
			});
		});
	}
 
	/**
	 *   
	 *
	 * @return {Promise}
	 */
	execute () {
		this._isRun = true;
 
		if (this._queue.length === 0) {
			return Promise.resolve([]);
		}
 
		const promises = [];
		const queues = this._queue;
 
		while (queues.length !== 0) {
			const tasks = queues.splice(0, 25);
 
			promises.push(
				this.vk.api.execute({
					code: getCode(tasks)
				})
				.then((result) => {
					resolvePromisesTask(tasks, result);
 
					return result.response;
				})
				.catch((error) => {
					for (const task of tasks) {
						task.reject(error);
					}
 
					throw error;
				})
			);
		}
 
		return Promise.all(promises)
		.then((responses) => Array.prototype.concat.apply([], responses));
	}
 
	/**
	 *   execute -> then
	 *
	 * @param  {function} handler
	 *
	 * @return {Promise}
	 */
	then (handler) {
		return this.execute().then(handler);
	}
 
	/**
	 *   execute -> catch
	 *
	 * @param {function} handler
	 *
	 * @return {Promise}
	 */
	catch (handler) {
		return this.execute().catch(handler);
	}
}

 * @param {Array} queues
 *
 * @return {string}

	return getChainCode(queues.map((queue) => queue.method));
}
